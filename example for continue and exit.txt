//example for continue
#include <stdio.h>

int main()
{
    int i;
    for(i=1;i<=5;i++)
{
    if(i==3)
    continue;
    printf("%d",i);
    
}
return 0;
}

//output:-  1 2 4 5






//example for exit
#include <stdio.h>

int main()
{
    int i;
    for(i=1;i<=5;i++)
{
    if(i==3)
    exit(0);
    printf("%d",i);
    
}
return 0;
}
